ComputeCluster 規格與驅動手冊 v4

適用檔案：simulator/include/ComputeCluster.hpp

1. 模組定位

ComputeCluster 封裝 ScratchpadMemory（SPM）、HybridDataDeliverUnit（HDDU）、NetworkOnChip（NoC），對外提供：

  • AXI4-Lite 64-bit Data Slave：對接 SPM DMA（preload / readback）
  • AHB-Lite 32-bit Command/MMIO Slave：SPM config/PMU + HDDU MMIO + NoC command
  • interrupt_o：來自 HDDU，受 power_enable_i gate

3. Micro Architecture

3.1 內部連線重點

  1. SPM DMA path：外部 AXI4-Lite data slave 信號經 comb_power_and_wiring power gate 後直連 SPM DMA 介面；power off 時 AWREADY=WREADY=ARREADY=0，BRESP=RRESP=DECERR
  2. HDDU ↔ SPM：HDDU 四個 SPM port 直連 SPM 四個 noc-side port（SPM_NUM_NOC_CHANNEL=4）；SPM_ARB_POLICY 為保留欄位
  3. HDDU ↔ NoC（VR 型別介面）：hddu.noc_ps/pd/pli_out → noc.noc_ps/pd/pli_in；hddu.noc_plo_out ↔ noc.noc_plo_in/out；PLO request 完全由 HDDU PLO channel 驅動
  4. SPM PMU：SPM 輸出 pmu_cycle_cnt_o、pmu_arb_stall_cnt_o、pmu_credit_stall_cnt_o（64-bit）及 pmu_port_txn_cnt_o[0..3]，由 seq_ahb_ctrl 讀出供 AHB 回傳

3.2 Power / Reset 行為

local_reset_n = reset_n && power_enable_i
  • power_enable_i=0：SPM/HDDU/NoC 全部等效 reset
  • AXI4-Lite：AWREADY=WREADY=ARREADY=0；BRESP=RRESP=2'b11（DECERR）；RDATA=0
  • AHB-Lite：hready_o=0，hrdata_o=0
  • interrupt_o=0

4.3 AHB-Lite Command/MMIO Slave（32-bit）

PortDirWidth說明
hsel_iin1Slave select
haddr_iin32位址（Address phase）
hwrite_iin11=write
htrans_iin2IDLE=0, BUSY=1, NONSEQ=2, SEQ=3
hsize_iin3傳輸寬度（保留）
hburst_iin3Burst 型態（保留）
hprot_iin4保護控制（保留）
hready_iin1前一個 slave ready（pipeline）
hwdata_iin32寫入資料（Data phase，比 address late 1 cycle）
hready_oout1= power_enable_i（無 wait states）
hresp_oout1= 0（OKAY）
hrdata_oout32讀回資料（power off → 0）
AHB-Lite 2-Cycle Pipeline：
Cycle N（Address Phase）：hsel && htrans[1]=1 && hready_i → 鎖存 haddr / hwrite / hsize
Cycle N+1（Data Phase）：根據鎖存位址執行讀寫；hwdata 在此週期有效；讀回值更新 ahb_rdata_reg，次週期輸出

5. MMIO Map（AHB 位址）

5.1 SPM config/PMU window（0x0000 ~ 0x00FF）

AddrNameRW說明
0x0000SPM_CONFIG_MAPR/W[7:0]：4 ports × 2-bit group mapping（config_map_i）
0x0004SPM_CONFIG_UPDATEWbit0=1 觸發 config_update_i 一拍 pulse（read 回傳 0）
0x0008SPM_ARB_POLICYR/Wbit0：保留，寫入不影響 SPM 仲裁行為
0x000CSPM_PMU_CTRLWbit0=1 觸發 PMU counter reset 一拍 pulse（read 回傳 0）
0x0010SPM_PMU_CYCLE_CNT_LORpmu_cycle_cnt_o[31:0]
0x0014SPM_PMU_CYCLE_CNT_HIRpmu_cycle_cnt_o[63:32]
0x0018SPM_PMU_ARB_STALL_LORpmu_arb_stall_cnt_o[31:0]
0x001CSPM_PMU_ARB_STALL_HIRpmu_arb_stall_cnt_o[63:32]
0x0020SPM_PMU_CREDIT_STALL_LORpmu_credit_stall_cnt_o[31:0]
0x0024SPM_PMU_CREDIT_STALL_HIRpmu_credit_stall_cnt_o[63:32]
0x0040 + N×8SPM_PMU_PORTn_TXN_LORpmu_port_txn_cnt_o[N][31:0]（N=0..3）
0x0044 + N×8SPM_PMU_PORTn_TXN_HIRpmu_port_txn_cnt_o[N][63:32]（N=0..3）

5.2 HDDU window（0x1000 ~ 0x1FFF）

直接 passthrough 到 HDDU MMIO：hddu_mmio_addr = haddr - 0x1000。範例：cluster 0x1800 → HDDU 0x800（HDDU_CTRL）。詳細欄位請參考 HDDU.md 第 5 節。

5.3 NoC command window（0x2000 ~ 0x20FF）

AddrNameRW說明
0x2000NOC_CMD_DATAR/WWrite：發送一拍 command_mode=1 + command_data=hwdata，更新 noc_last_cmd_reg；Read：回傳 noc_last_cmd_reg
PLO request/response 路徑由 HDDU PLO channel 直接驅動，不經 NoC MMIO window。

5.4 命令分流規則

haddr 依區間分流（seq_ahb_ctrl Data Phase 執行）：

地址範圍目標
0x0000 ~ 0x00FFSPM config / PMU 暫存器
0x1000 ~ 0x1FFFHDDU MMIO passthrough
0x2000 ~ 0x20FFNoC command sideband
其他讀回 0，寫入忽略

6. 驅動共用工具（AGU / NoC command / PE program）

// Platform-side helpers（請依實際介面實作）
void ahb_write(uint32_t addr, uint32_t data);   // AHB-Lite 32-bit write（2-cycle pipeline）
uint32_t ahb_read(uint32_t addr);               // AHB-Lite 32-bit read
void axi_write64(uint32_t addr, uint64_t data); // AXI4-Lite 64-bit DMA write
uint64_t axi_read64(uint32_t addr);             // AXI4-Lite 64-bit DMA read

// ---- Cluster / HDDU base ----
static constexpr uint32_t CLUSTER_HDDU_BASE = 0x1000;
static constexpr uint32_t CLUSTER_NOC_CMD   = 0x2000;

static constexpr uint32_t AGU_BANK_STRIDE = 0x100;
static constexpr uint32_t AGU_PS  = 0;
static constexpr uint32_t AGU_PD  = 1;
static constexpr uint32_t AGU_PLI = 2;
static constexpr uint32_t AGU_PLO = 3;

// ---- AGU bank offset（AddressGenerateUnit::RegOffset）----
static constexpr uint32_t REG_BASE_ADDR   = 0x00;
static constexpr uint32_t REG_BASE_ADDR_H = 0x04;
static constexpr uint32_t REG_ITER01      = 0x08;
static constexpr uint32_t REG_ITER23      = 0x0C;
static constexpr uint32_t REG_STRIDE0     = 0x10;
static constexpr uint32_t REG_STRIDE1     = 0x14;
static constexpr uint32_t REG_STRIDE2     = 0x18;
static constexpr uint32_t REG_STRIDE3     = 0x1C;
static constexpr uint32_t REG_CTRL        = 0x20; // bit0=start, bit3=ultra
static constexpr uint32_t REG_LANE_CFG    = 0x28;
static constexpr uint32_t REG_TAG_BASE    = 0x40;
static constexpr uint32_t REG_TAG_STRIDE0 = 0x44;
static constexpr uint32_t REG_TAG_STRIDE1 = 0x48;
static constexpr uint32_t REG_TAG_CTRL    = 0x4C;
static constexpr uint32_t REG_MASK_CFG    = 0x54;

// ---- HDDU global offset（相對於 0x1000）----
static constexpr uint32_t HDDU_CTRL       = 0x800; // bit0=soft-reset, bit1=start_all, bit2=stop_all
static constexpr uint32_t HDDU_STATUS     = 0x804; // bit1=any_busy, bit2=done(!any_busy), bit4=err
static constexpr uint32_t HDDU_PLANE_EN   = 0x808; // bit0~3 = PS/PD/PLI/PLO enable
static constexpr uint32_t HDDU_PLANE_MODE = 0x80C; // software mode flag（硬體不解碼）
static constexpr uint32_t HDDU_ARB_POLICY = 0x818; // 保留欄位

// ---- NoC command ----
enum message_command_t : uint32_t {
	CMD_RESET = 0,
	CMD_INIT = 1,
	CMD_LOAD_PROGRAM = 2,
	CMD_STOP_PE = 3,
	CMD_START_PE = 4,
	CMD_NOC_SCAN_CHAIN = 8,
};

static constexpr uint32_t PE_ROUTER_IM_ADDR_OFFSET = 4;
static constexpr uint32_t PE_ROUTER_IM_DATA_OFFSET = 16;
static constexpr uint32_t PE_ROUTER_IM_ADDR_MASK   = 0xFFF;
static constexpr uint32_t PE_ROUTER_IM_DATA_MASK   = 0xFFFF;

uint32_t pack_noc_cmd(message_command_t cmd, uint32_t param) {
	return (param & 0xFFFFFFF0u) | (static_cast(cmd) & 0x0Fu);
}

uint32_t pack_scan_chain(uint8_t ps_id, uint8_t pd_id, uint8_t pli_id, uint8_t plo_id,
						 uint8_t route_mode, bool enable) {
	uint32_t v = 0;
	v |= (static_cast(ps_id)  & 0x3Fu) << 4;
	v |= (static_cast(pd_id)  & 0x3Fu) << 10;
	v |= (static_cast(pli_id) & 0x3Fu) << 16;
	v |= (static_cast(plo_id) & 0x3Fu) << 22;
	v |= (static_cast(route_mode) & 0x03u) << 28;
	v |= (enable ? 1u : 0u) << 30;
	return pack_noc_cmd(CMD_NOC_SCAN_CHAIN, v);
}

uint32_t pack_load_program(uint16_t im_addr_bytes, uint16_t inst16) {
	uint32_t p = 0;
	p |= (static_cast(im_addr_bytes) & PE_ROUTER_IM_ADDR_MASK) << PE_ROUTER_IM_ADDR_OFFSET;
	p |= (static_cast(inst16)        & PE_ROUTER_IM_DATA_MASK) << PE_ROUTER_IM_DATA_OFFSET;
	return pack_noc_cmd(CMD_LOAD_PROGRAM, p);
}

void noc_cmd_write(uint32_t packed_cmd) {
	ahb_write(CLUSTER_NOC_CMD, packed_cmd); // 寫 0x2000，ComputeCluster 打一拍 command_mode=1
}

struct AguCfg {
	uint32_t base_addr;
	uint16_t iter0, iter1, iter2, iter3;
	int32_t stride0, stride1, stride2, stride3;
	uint32_t lane_cfg;
	uint32_t tag_base, tag_stride0, tag_stride1, tag_ctrl;
	uint32_t mask_cfg;
	bool ultra;
};

void cfg_agu(uint32_t bank, const AguCfg& c) {
	const uint32_t B = CLUSTER_HDDU_BASE + bank * AGU_BANK_STRIDE;
	ahb_write(B + REG_BASE_ADDR,   c.base_addr);
	ahb_write(B + REG_BASE_ADDR_H, 0);
	ahb_write(B + REG_ITER01,      (static_cast<uint32_t>(c.iter1) << 16) | c.iter0);
	ahb_write(B + REG_ITER23,      (static_cast<uint32_t>(c.iter3) << 16) | c.iter2);
	ahb_write(B + REG_STRIDE0,     static_cast<uint32_t>(c.stride0));
	ahb_write(B + REG_STRIDE1,     static_cast<uint32_t>(c.stride1));
	ahb_write(B + REG_STRIDE2,     static_cast<uint32_t>(c.stride2));
	ahb_write(B + REG_STRIDE3,     static_cast<uint32_t>(c.stride3));
	ahb_write(B + REG_LANE_CFG,    c.lane_cfg);
	ahb_write(B + REG_TAG_BASE,    c.tag_base);
	ahb_write(B + REG_TAG_STRIDE0, c.tag_stride0);
	ahb_write(B + REG_TAG_STRIDE1, c.tag_stride1);
	ahb_write(B + REG_TAG_CTRL,    c.tag_ctrl);
	ahb_write(B + REG_MASK_CFG,    c.mask_cfg);
	ahb_write(B + REG_CTRL,        c.ultra ? (1u << 3) : 0u); // ultra flag 先配，不在此 start
}

7. 完整驅動範例：Conv2D（四 AGU + scan-chain + PE program）

// 1) power / reset
power_enable_i = 1;
reset_n = 0; wait_cycles(4);
reset_n = 1; wait_cycles(4);

// 2) SPM mapping（AHB-Lite）
ahb_write(0x0000, 0xE4);  // 4-port group mapping
ahb_write(0x0004, 0x1);   // trigger config_update pulse

// 3) preload data（AXI4-Lite DMA，位址單位 byte）
for (uint32_t i = 0; i < weight_words; ++i) axi_write64(weight_base + i*8, weight_buf[i]);
for (uint32_t i = 0; i < ifmap_words;  ++i) axi_write64(ifmap_base  + i*8, ifmap_buf[i]);
for (uint32_t i = 0; i < pli_words;    ++i) axi_write64(pli_base    + i*8, pli_init[i]);

// 4) AGU 全設定（PS / PD / PLI / PLO）
AguCfg ps_cfg  = {ps_base,  ps_i0,  ps_i1,  ps_i2,  ps_i3,  ps_s0,  ps_s1,  ps_s2,  ps_s3,  0, ps_tag_base,  ps_t0,  ps_t1,  ps_tctrl,  0xF, false};
AguCfg pd_cfg  = {pd_base,  pd_i0,  pd_i1,  pd_i2,  pd_i3,  pd_s0,  pd_s1,  pd_s2,  pd_s3,  0, pd_tag_base,  pd_t0,  pd_t1,  pd_tctrl,  0xF, false};
AguCfg pli_cfg = {pli_base, pli_i0, pli_i1, pli_i2, pli_i3, pli_s0, pli_s1, pli_s2, pli_s3, 0, pli_tag_base, pli_t0, pli_t1, pli_tctrl, 0xF, false};
AguCfg plo_cfg = {plo_base, plo_i0, plo_i1, plo_i2, plo_i3, plo_s0, plo_s1, plo_s2, plo_s3, 0, plo_tag_base, plo_t0, plo_t1, plo_tctrl, 0xF, false};

cfg_agu(AGU_PS,  ps_cfg);
cfg_agu(AGU_PD,  pd_cfg);
cfg_agu(AGU_PLI, pli_cfg);
cfg_agu(AGU_PLO, plo_cfg);

// 5) HDDU global 設定
ahb_write(CLUSTER_HDDU_BASE + HDDU_CTRL,       (1u << 0)); // soft-reset / clear FIFO
ahb_write(CLUSTER_HDDU_BASE + HDDU_PLANE_EN,   0xF);       // PS/PD/PLI/PLO 全開
ahb_write(CLUSTER_HDDU_BASE + HDDU_PLANE_MODE, 0x1);       // conv2d mode flag

// 6) NoC 初始化 + scan-chain（注意：依實作需 reverse 順序 shift）
noc_cmd_write(pack_noc_cmd(CMD_RESET, 0));
noc_cmd_write(pack_noc_cmd(CMD_INIT,  0x12340000));

for (int i = static_cast(scan_chain_words.size()) - 1; i >= 0; --i) {
	// 每個 word 欄位: ps[9:4], pd[15:10], pli[21:16], plo[27:22], mode[29:28], en[30]
	noc_cmd_write(pack_noc_cmd(CMD_NOC_SCAN_CHAIN, scan_chain_words[i]));
}

// 7) 載入 PE program（每筆 16-bit 指令）
for (uint32_t pc = 0; pc < pe_program_words; ++pc) {
	uint16_t inst = pe_program[pc];
	uint16_t im_addr = static_cast(pc * 2); // byte address
	noc_cmd_write(pack_load_program(im_addr, inst));
}

// 8) 啟動 PE 與 HDDU
noc_cmd_write(pack_noc_cmd(CMD_START_PE, 0));
ahb_write(CLUSTER_HDDU_BASE + HDDU_CTRL, (1u << 1)); // start_all

// 9) 等待完成（輪詢 HDDU_STATUS）
// bit1=any_busy, bit4=err；!any_busy → done
while (true) {
	uint32_t st = ahb_read(CLUSTER_HDDU_BASE + HDDU_STATUS);
	if (st & (1u << 4)) { /* error */ break; }
	if (!(st & (1u << 1))) { /* !any_busy → done */ break; }
}

// 10) 讀回 ofmap（AXI4-Lite DMA）
for (uint32_t i = 0; i < ofmap_words; ++i)
	ofmap_buf[i] = axi_read64(ofmap_base + i * 8);

建議同步讀取 HDDU counters（0x1828 TX_PKT / 0x182C TX_BYTE / 0x1830 RX_BYTE / 0x1834 STALL）做吞吐與背壓分析。

9. 除錯與驗證建議

建議分層驗證順序：

  1. AXI4-Lite data slave ↔ SPM 基本讀寫
  2. AHB-Lite SPM mapping/更新脈衝行為（0x0000 / 0x0004）
  3. AHB-Lite HDDU MMIO + AGU 設定讀回（0x1xxx）
  4. AHB-Lite NoC command（0x2000）
  5. 全鏈路 Conv2D / GEMM end-to-end

建議監控：

  • HDDU counters: 0x1828 COUNTER_TX_PKT / 0x182C TX_BYTE / 0x1830 RX_BYTE / 0x1834 STALL
  • HDDU status: 0x1804（bit1=any_busy, bit4=err, bit3=stall）
  • SPM PMU: 0x0010~0x0024 cycle/arb_stall/credit_stall；0x0040~0x005C port txn counters
  • interrupt_o 脈衝（來自 HDDU status.err 或 status.done）
